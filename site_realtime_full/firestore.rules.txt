
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() { return request.auth != null; }

    match /users/{uid} {
      allow read: if true;
      allow create, update: if signedIn() && request.auth.uid == uid;
    }
    match /user_codes/{id6} {
      allow read: if true;
      allow create, delete: if signedIn();
    }

    match /friendRequests/{rid} {
      allow read: if signedIn();
      allow create: if signedIn() && request.resource.data.from == request.auth.uid;
      allow update: if signedIn() && (request.auth.uid == resource.data.to);
    }

    match /chats/{chatId} {
      allow read: if signedIn() && (request.auth.uid in resource.data.memberIds);
      allow create: if signedIn();
      allow update: if signedIn();
      allow delete: if signedIn() && resource.data.isGroup == true && resource.data.ownerId == request.auth.uid;

      match /messages/{mid} {
        allow read: if signedIn() && (request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.memberIds);
        allow create: if signedIn() && (request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.memberIds);
        // 允许消息发送者或群管理者撤回（更新）
        allow update: if signedIn() && (
          request.auth.uid == resource.data.senderId ||
          (get(/databases/$(database)/documents/chats/$(chatId)).data.members[request.auth.uid] in ['owner','admin'])
        );
      }
      match /announcements/{aid} {
        allow read: if signedIn() && (request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.memberIds);
        allow create: if signedIn() && (
          get(/databases/$(database)/documents/chats/$(chatId)).data.members[request.auth.uid] in ['owner','admin']
        );
      }
    }

    match /comments/{cid} {
      allow read: if true;
      allow create: if signedIn();
      // 仅评论作者本人可删除/修改
      allow update, delete: if signedIn() && request.auth.uid == resource.data.userId;
    }
  }
}
